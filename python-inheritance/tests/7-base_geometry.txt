BaseGeometry = __import__('7-base_geometry').BaseGeometry

# Object yaradılır
bg = BaseGeometry()

# area() test
try:
    bg.area()
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

# integer_validator() testləri

# ✔ keçməli
bg.integer_validator("age", 1)

# ❗ ValueError
try:
    bg.integer_validator("age", 0)
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

try:
    bg.integer_validator("age", -4)
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

# ❗ TypeError – int deyil
values = ["4", (4,), [3], True, {3, 4}, None]

for v in values:
    try:
        bg.integer_validator("age", v)
    except Exception as e:
        print("[{}] {}".format(e.__class__.__name__, e))
